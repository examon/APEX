19 bool APEXPass::runOnModule(Module &M) {
20  logPrintUnderline("APEXPass START.");
21 
22 
23  if (VERBOSE_DEBUG) {
24  logPrintUnderline("Initial module dump.");
25  logDumpModule(M);
26  }
27 
28  logPrintUnderline("Parsing command line arguments.");
29  moduleParseCmdLineArgsOrDie();
30 
31  logPrintUnderline("Locating target instructions.");
32  moduleFindTargetInstructionsOrDie(M, ARG_FILE, ARG_LINE);
33 
34  logPrintUnderline("Collecting protected functions.");
35  collectProtectedFunctions(M);
36 
37  logPrintUnderline(
38  "Initializing dg. Calculating control and data dependencies.");
39  dgInit(M);
40 
41  logPrintUnderline("Extracting data from dg. Building apex dependency graph.");
42  apexDgInit();
43 
44  if (VERBOSE_DEBUG) {
45  logPrintUnderline("Printing data dependencies from apex_dg.");
46  apexDgPrintDependenciesCompact();
47  }
48 
49  logPrintUnderline("Constructing function dependency blocks.");
50  apexDgComputeFunctionDependencyBlocks(M);
51 
52  if (VERBOSE_DEBUG) {
53  logPrintUnderline("Printing calculated function dependency blocks.");
54  apexDgPrintFunctionDependencyBlocks();
55  }
56 
57  logPrintUnderline("Constructing dependency blocks to functions call graph.");
58  apexDgConstructBlocksFunctionsCallgraph();
59 
60  if (VERBOSE_DEBUG) {
61  logPrintUnderline("Printing dependency block to functions call graph.");
62  apexDgPrintBlocksFunctionsCallgraph();
63  }
64 
65  logPrintUnderline("Finding path from @" + source_function_id_ + " to @" +
66  target_function_id_ + ".");
67  findPath(M);
68 
69  if (VERBOSE_DEBUG) {
70  logPrintUnderline("Printing path from @" + source_function_id_ + " to @" +
71  target_function_id_ + ".");
72  printPath();
73  }
74 
75  logPrintUnderline("Removing functions and dependency blocks that do not "
76  "affect calculated path.");
77  removeUnneededStuff(M);
78 
79  logPrintUnderline("Injecting exit and extract calls.");
80  moduleInjectExitExtract(M);
81 
82  logPrintUnderline("Stripping debug symbols from every function in module.");
83  stripAllDebugSymbols(M);
84 
85  if (VERBOSE_DEBUG) {
86  logPrintUnderline("Final module dump.");
87  logDumpModule(M);
88  }
89 
90  logPrintUnderline("APEXPass END.");
91  return true;
92 }

153 void APEXPass::findPath(const Module &M) {
154  // @block_path is used to store path from @source_function to each node.
155  std::map<DependencyBlock *, std::vector<DependencyBlock>> block_path;
156  {
157  std::vector<DependencyBlock *> queue;
158  std::vector<DependencyBlock *> visited;
159 
160  // Initially push blocks from source function into the @queue.
161  for (auto &block :
162  function_dependency_blocks_[M.getFunction(source_function_id_)]) {
163  queue.push_back(&block);
164  }
165 
166  while (false == queue.empty()) {
167  const auto current_ptr = queue.back();
168  queue.pop_back();
169  visited.push_back(current_ptr);
170 
171  // Find if there are calls to other functions that come from the current
172  // block.
173  for (const auto &called_function :
174  blocks_functions_callgraph_[*current_ptr]) {
175  // Get blocks for those functions that are called.
176  for (auto &neighbour_block :
177  function_dependency_blocks_[called_function]) {
178  // What we do here is that we copy the path from the current
179  // block, append current node to it and assign this path to
180  // each neighbour of current.
181  block_path[&neighbour_block] = block_path[current_ptr];
182  block_path[&neighbour_block].push_back(*current_ptr);
183 
184  bool block_already_visited = false;
185  for (const auto &visited_block_ptr : visited) {
186  if (visited_block_ptr == &neighbour_block) {
187  block_already_visited = true;
188  break;
189  }
190  }
191  // Add non-visited blocks from called function into the queue.
192  if (false == block_already_visited) {
193  queue.push_back(&neighbour_block);
194  }
195  }
196  }
197  }
198  }
199 
200  // Finding target block. That is, block where target instruction are.
201  // We use heuristic and take last instruction from the @target_instructions_.
202  // The reason is that inside @target_instructions_ may be more instructions
203  // than present in target block (especially at the beginning), so
204  DependencyBlock *target_block = nullptr;
205  for (auto &block :
206  function_dependency_blocks_[M.getFunction(target_function_id_)]) {
207  for (const auto &node_ptr : block) {
208  if (target_instructions_.back() == node_ptr->getValue()) {
209  // Check the target line, just in case.
210  if (std::to_string(
211  target_instructions_.back()->getDebugLoc().getLine()) ==
212  ARG_LINE) {
213  target_block = &block;
214  }
215  }
216  }
217  }
218  if (nullptr == target_block || target_block->empty()) {
219  logPrint("ERROR: Could not find target block! Make sure target "
220  "instructions are not dead code (e.g. in the uncalled function");
221  exit(FATAL_ERROR);
222  }
223 
224  logPrint("Reconstructing block path:");
225  for (const auto &bp : block_path) {
226  if (bp.first == target_block) {
227  path_ = bp.second;
228  }
229  }
230  path_.push_back(*target_block);
231  logPrint("- done");
232 }

250 void APEXPass::dgInit(Module &M) {
251  // The reasoning why everything below needs to be here:
252  //
253  // In order to get data dependencies, I've replicated
254  // what llvm-dg-dump tool is doing, so for details, check:
255  // dg/tools/llvm-dg-dump.cpp
256  LLVMPointerAnalysis *pta = new LLVMPointerAnalysis(&M);
257  pta->run<analysis::pta::PointsToFlowInsensitive>();
258  dg_.build(&M, pta);
259  analysis::rd::LLVMReachingDefinitions rda(&M, pta);
260  rda.run<analysis::rd::ReachingDefinitionsAnalysis>();
261  // rda.run<analysis::rd::SemisparseRda>(); // This is alternative to above
262  // ^^
263  LLVMDefUseAnalysis dua(&dg_, &rda, pta);
264  dua.run();
265  dg_.computeControlDependencies(CD_ALG::CLASSIC);
266 
267  logPrint("- done");
268 }

278 void APEXPass::apexDgInit() {
279  const std::map<llvm::Value *, LLVMDependenceGraph *> &CF =
280  dg::getConstructedFunctions(); // Need to call this (dg reasons).
281 
282  logPrint("Initializing @apex_dg structures:");
283  for (auto &function_dg : CF) {
284  APEXDependencyFunction apex_function;
285  apex_function.value = function_dg.first;
286 
287  for (auto &value_block : function_dg.second->getBlocks()) {
288  // We do not care about blocks, we care about function:node
289  // relationship.
290  for (auto &node : value_block.second->getNodes()) {
291  // Create node (AKA instruction) structure and fill it with
292  // control & data dependencies to other nodes.
293  APEXDependencyNode apex_node;
294  apex_node.node = node;
295  apex_node.value = node->getValue();
296 
297  // Store data & control dependencies from @node to @apex_node.
298  apexDgGetBlockNodeInfo(apex_node, node);
299 
300  apex_function.nodes.push_back(apex_node);
301  }
302  }
303  apex_dg_.functions.push_back(apex_function);
304  }
305  logPrint("- done");
306 
307  logPrint("\nStoring data and control dependencies into @apex_dg structures:");
308  for (auto &apex_dg_function : apex_dg_.functions) {
309  for (auto &node : apex_dg_function.nodes) {
310  // Data & reverse data dependencies.
311  apex_dg_.node_data_dependencies_map[node.node] = node.data_dependencies;
312  apex_dg_.node_rev_data_dependencies_map[node.node] =
313  node.rev_data_dependencies;
314 
315  // Control & reverse control dependencies.
316  apex_dg_.node_control_dependencies_map[node.node] =
317  node.control_depenencies;
318  apex_dg_.node_rev_control_dependencies_map[node.node] =
319  node.rev_control_depenencies;
320 
321  // Map function to node.
322  apex_dg_.node_function_map[node.node] = &apex_dg_function;
323  }
324  }
325  logPrint("- done");
326 }

330 void APEXPass::apexDgGetBlockNodeInfo(APEXDependencyNode &apex_node,
331  LLVMNode *node) {
332  for (auto i = node->control_begin(), e = node->control_end(); i != e; ++i) {
333  LLVMNode *cd_node = *i;
334  apex_node.control_depenencies.push_back(cd_node);
335  }
336 
337  for (auto i = node->rev_control_begin(), e = node->rev_control_end(); i != e;
338  ++i) {
339  LLVMNode *rev_cd_node = *i;
340  apex_node.rev_control_depenencies.push_back(rev_cd_node);
341  }
342 
343  for (auto i = node->data_begin(), e = node->data_end(); i != e; ++i) {
344  LLVMNode *dd_node = *i;
345  apex_node.data_dependencies.push_back(dd_node);
346  }
347 
348  for (auto i = node->rev_data_begin(), e = node->rev_data_end(); i != e; ++i) {
349  LLVMNode *rev_dd_node = *i;
350  apex_node.rev_data_dependencies.push_back(rev_dd_node);
351  }
352 }

380  {
381  std::vector<LLVMNode *> visited;
382  std::vector<LLVMNode *> queue;
383  queue.push_back(&node);
384  while (false == queue.empty()) {
385  // Pop @curr and investigate it first.
386  LLVMNode *curr = queue.back();
387  queue.pop_back();
388 
389  bool already_visited = false;
390  for (LLVMNode *visited_node : visited) {
391  if (curr == visited_node) {
392  already_visited = true;
393  break;
394  }
395  }
396  if (already_visited) {
397  continue;
398  }
399  visited.push_back(curr);
400 
401  // Store @curr if it is not already stored.
402  bool new_dependency = true;
403  for (LLVMNode *node : data_dependencies) {
404  if (curr == node) {
405  new_dependency = false;
406  }
407  }
408  if (new_dependency && curr != &node) {
409  data_dependencies.push_back(curr);
410  }
411 
412  // Add neighbours to the queue.
413  for (LLVMNode *neighbor : apex_dg_.node_data_dependencies_map[curr]) {
414  queue.push_back(neighbor);
415  }
416  }
417  }
418  {
419  std::vector<LLVMNode *> queue;
420  queue.push_back(&node);
421  while (false == queue.empty()) {
422  // Pop @curr and investigate it first.
423  LLVMNode *curr = queue.back();
424  queue.pop_back();
425 
426  std::vector<LLVMNode *> visited;
427  bool already_visited = false;
428  for (LLVMNode *visited_node : visited) {
429  if (curr == visited_node) {
430  already_visited = true;
431  break;
432  }
433  }
434  if (already_visited) {
435  continue;
436  }
437  visited.push_back(curr);
438 
439  // Store @curr if it is not already stored.
440  bool new_dependency = true;
441  for (LLVMNode *node : rev_data_dependencies) {
442  if (curr == node) {
443  new_dependency = false;
444  }
445  }
446  if (new_dependency && curr != &node) {
447  rev_data_dependencies.push_back(curr);
448  }
449 
450  // Add neighbours to the queue.
451  for (LLVMNode *neighbor : apex_dg_.node_rev_data_dependencies_map[curr]) {
452  queue.push_back(neighbor);

485 void APEXPass::apexDgComputeFunctionDependencyBlocks(const Module &M) {
486  logPrint("Constructing dependency blocks:");
487 
488  for (auto &F : M) {
489  logPrintDbg("Constructing dependency blocks for: " +
490  F.getGlobalIdentifier());
491 
492  if (functionIsProtected(&F)) {
493  logPrintDbg("- function is protected");
494  continue;
495  }
496 
497  bool function_in_apex_dg = false;
498  for (APEXDependencyFunction &apex_fcn : apex_dg_.functions) {
499  if (&F == apex_fcn.value) {
500  function_in_apex_dg = true;
501  }
502  }
503  if (false == function_in_apex_dg) {
504  logPrintDbg("- function not in @apex_dg (probably not used, etc.)");
505  continue;
506  }
507 
508  std::vector<std::vector<LLVMNode *>> function_blocks;
509  unsigned num_fcn_instructions = 0;
510 
511  for (auto &BB : F) {
512 
513  std::vector<LLVMNode *> visited;
514  for (auto &I : BB) {
515  num_fcn_instructions++;
516 
517  std::vector<LLVMNode *> new_block;
518  APEXDependencyNode apex_node;
519  apexDgGetNodeOrDie(apex_dg_, &I, apex_node);
520 
521  bool go_to_next_instruction = false;
522 
523  // Check if @I is already in some stored block.
524  for (auto &block : function_blocks) {
525  for (LLVMNode *dep_block_node : block) {
526  if (apex_node.node == dep_block_node) {
527  // @apex_node is already in some @block.
528  // Go investigate directly next instruction.
529  go_to_next_instruction = true;
530  }
531  }
532  }
533  if (go_to_next_instruction) {
534  continue;
535  }
536 
537  std::vector<LLVMNode *> queue;
538  queue.push_back(apex_node.node);
539 
540  while (false == queue.empty()) {
541  LLVMNode *current = queue.back();
542  queue.pop_back();
543 
544  bool already_visited = false;
545  for (LLVMNode *visited_node : visited) {
546  if (current == visited_node) {
547  already_visited = true;
548  break;
549  }
550  }
551  if (already_visited) {
552  continue;
553  }
554 
555  visited.push_back(current);
556 
557  // Get dependency chain for @current;
558  std::vector<LLVMNode *> data_dependencies;
559  std::vector<LLVMNode *> rev_data_dependencies;
560  apexDgFindDataDependencies(*apex_node.node, data_dependencies,
561  rev_data_dependencies);
562  // I think it is enough to consider only data dependencies and
563  // do not care about reverse data dependencies.
564  // Reverse data dependencies contain instructions that are outside
565  // current function and that created problems.
566  std::vector<LLVMNode *> neighbours = data_dependencies;
567 
568  bool current_already_stored = false;
569 
570  for (LLVMNode *neighbour : neighbours) {
571 
572  bool neighbour_already_stored = false;
573  for (auto &block : function_blocks) {
574 
575  for (LLVMNode *node : block) {
576  if (neighbour == node) {
577  neighbour_already_stored = true;
578 
579  for (auto &b : function_blocks) {
580  for (LLVMNode *n : b) {
581  if (current == n) {
582  current_already_stored = true;
583  }
584  }
585  }
586 
587  if (false == current_already_stored) {
588  block.push_back(current);
589  current_already_stored = true;
590  }
591  }
592  }
593  }
594  if (false == neighbour_already_stored) {
595  queue.push_back(neighbour);
596  }
597  }
598 
599  if (false == current_already_stored) {
600  new_block.push_back(current);
601  }
602  }
603 
604  if (new_block.size() > 0) {
605  function_blocks.push_back(new_block);
606  }
607  }
608  }
609 
610  // Checking if we did not miss any instruction.
611  unsigned num_dep_blocks_instructions = 0;
612  for (auto &block : function_blocks) {
613  num_dep_blocks_instructions += block.size();
614  }
615 // assert(num_fcn_instructions == num_dep_blocks_instructions);
616  logPrintDbg("- done: " + std::to_string(num_fcn_instructions) +
617  " instructions in blocks");
618 
619  // Everything should be OK, store computed blocks into storage.
620  function_dependency_blocks_[&F] = function_blocks;
621  }
622  logPrint("- done");
623 
624  // Instructions in stored in the @new_blocks are not in the same order
625  // as they are in the @F.
626  // We need to put them in the correct order and store that result back
627  // to the @function_dependency_blocks.
628  logPrint("\nSorting function dependency blocks:");
629  std::map<const Function *, std::vector<std::vector<LLVMNode *>>>
630  function_dependency_blocks_sorted;
631  for (const auto &function_blocks : function_dependency_blocks_) {
632  for (const auto &block : function_blocks.second) {
633  std::vector<LLVMNode *> new_block_sorted;
634  for (const auto &BB : *(function_blocks.first)) {
635  for (const auto &I : BB) {
636  for (const auto node_ptr : block) {
637  if (node_ptr->getValue() == &I) {
638  new_block_sorted.push_back(node_ptr);
639  }
640  }
641  }
642  }
643  // Make sure we have not missed any instruction and sorted block has
644  // every instruction that unsorted before storing @new_block_sorted.
645 // assert(new_block_sorted.size() == block.size());
646  function_dependency_blocks_sorted[function_blocks.first].push_back(
647  new_block_sorted);
648  }
649  }
650  function_dependency_blocks_ = function_dependency_blocks_sorted;
651  logPrint("- done");
652 }

682  // Store call edge from block to function.

684  functions.push_back(called_fcn);

687  }
688  }
689  }

727 void APEXPass::moduleFindTargetInstructionsOrDie(Module &M,
728  const std::string &file,
729  const std::string &line) {
730  logPrintDbg("ARG_FILE:" + ARG_FILE);
731  logPrintDbg("ARG_LINE:" + ARG_LINE);
732  for (const auto &F : M) {
733  for (const auto &BB : F) {
734  for (const auto &I : BB) {
735  if (auto const debug_info = I.getDebugLoc()) {
736  const std::string line = std::to_string(debug_info->getLine());
737  const std::string file = debug_info->getFilename();
738  // std::string dir = inst_loc_ptr->getDirectory();
739  I.dump();

770 void APEXPass::moduleInjectExitExtract(Module &M) {
771  logPrint("Supplied target instructions from: " +
772  target_instructions_.back()->getFunction()->getGlobalIdentifier());
773  for (const auto &target_instruction : target_instructions_) {
774  if (nullptr == target_instruction) {
775  continue;

779 
780  logPrint("\nSetting injection point:");
781  Instruction *injection_point =
782  const_cast<Instruction *>(target_instructions_.back());
783  logPrintFlat("-");
784  injection_point->dump();
785 
786  logPrint("\nInjecting call instruction to _apex_exit():");
787  {
788  // Create vector of types and put one integer into this vector.
789  std::vector<Type *> params_tmp = {Type::getInt32Ty(M.getContext())};
790  // We need to convert vector to ArrayRef.
791  ArrayRef<Type *> params = makeArrayRef(params_tmp);
792 
793  // Create function that returns void and has parameters that are
794  // specified in the @params (that is, one Int32).
795  FunctionType *fcn_type =
796  FunctionType::get(Type::getVoidTy(M.getContext()), params, false);
797 
798  // Load exit function into variable.
799  Constant *temp = M.getOrInsertFunction("_apex_exit", fcn_type);
800  if (nullptr == temp) {
801  logPrint("ERROR: lib_exit function is not in symbol table.");
802  exit(-1);
803  }
804  Function *_apex_exit_fcn = cast<Function>(temp);
805  logPrint("- loaded function: " + _apex_exit_fcn->getGlobalIdentifier());
806 
807  // Create exit call instruction.
808  Value *_apex_exit_fcn_arg_val =
809  ConstantInt::get(Type::getInt32Ty(M.getContext()), APEX_DONE);
810  ArrayRef<Value *> _apex_exit_fcn_params =
811  makeArrayRef(_apex_exit_fcn_arg_val);
812  // CallInst::Create build call instruction to @exit_fcn that has
813  // @_apex_exit_fcn_params. Empty string "" means that the
814  // @_apex_exit_call_inst will not have return variable.
815  CallInst *_apex_exit_call_inst =
816  CallInst::Create(_apex_exit_fcn, _apex_exit_fcn_params, "");
817 
818  // If the @insertion point is terminator, insert @_apex_exit_call_inst
819  // before it. Otherwise, basic blocks will not end with terminator.
820  // TODO: Will this inconsistency cause problems?
821  if (injection_point->isTerminator()) {
822  logPrintFlat("- @injection_point is terminator, inserting " +
823  _apex_exit_fcn->getGlobalIdentifier() + " before: ");
824  injection_point->dump();
825  _apex_exit_call_inst->insertBefore(injection_point);
826  } else {
827  logPrintFlat("- @injection_point is NOT terminator, inserting " +
828  _apex_exit_fcn->getGlobalIdentifier() + " after: ");
829  injection_point->dump();
830  _apex_exit_call_inst->insertAfter(injection_point);
831  }
832 
833  // Final check.
834  if (nullptr == _apex_exit_call_inst) {
835  logPrint("ERROR: could not create " +
836  _apex_exit_fcn->getGlobalIdentifier() + " call instruction.");
837  exit(-1);
838  }
839  logPrintFlat("- " + _apex_exit_fcn->getGlobalIdentifier() +
840  " call instruction created: ");
841  _apex_exit_call_inst->dump();
842 
843  // Set insertion point to the exit call we just inserted.
844  // All new instructions are going to be inserted before exit call.
845  injection_point = _apex_exit_call_inst;
846  }
847 
848  logPrint("\nInjecting call instruction to _apex_extract_int():");
849  {
850  // Load _apex_extract_int() from lib.c and save it.
851  Instruction *last_target =
852  const_cast<Instruction *>(target_instructions_.back());
853  std::vector<Type *> params_tmp = {PointerType::getInt32Ty(M.getContext())};
854  ArrayRef<Type *> params = makeArrayRef(params_tmp);
855  FunctionType *fcn_type =
856  FunctionType::get(Type::getVoidTy(M.getContext()), params, false);
857  Constant *temp = M.getOrInsertFunction("_apex_extract_int", fcn_type);
858  Function *_apex_extract_int_fcn = cast<Function>(temp);
859  logPrint("- loaded function: " +
860  _apex_extract_int_fcn->getGlobalIdentifier());
861 
862  logPrintFlat("- last of the target functions: ");
863  last_target->dump();
864 
865  if (false == isa<StoreInst>(last_target)) {
866  logPrint("ERROR: Invalid target instruction! It has to be StoreInst!");
867  exit(FATAL_ERROR);
868  }
869 
870  if (2 != last_target->getNumOperands()) {
871  logPrint("ERROR: Invalid target instruction! It has to have 2 operands!");
872  exit(FATAL_ERROR);
873  }
874 
875  // Example of instruction: store i32 %call, i32* %f, allign 4
876  // @_apex_extract_int_arg would be second operand: i32* %f
877  Value *_apex_extract_int_arg =
878  last_target->getOperand(last_target->getNumOperands() - 1);
879  logPrintFlat("- last of the target instructions, second op: ");
880  _apex_extract_int_arg->dump();
881 
882  // Now we need to load that pointer to i32 into classic i32.
883  // Because @lib_extract_int takes i32 instead of i32*.
884  LoadInst *_apex_extract_int_arg_loadinst = new LoadInst(
885  _apex_extract_int_arg, "_apex_extract_int_arg", injection_point);
886  logPrintFlat("- created load instruction: ");
887  _apex_extract_int_arg_loadinst->dump();
888 
889  // Now make callinst to _apex_extract_int with the correct argument
890  // (that is i32 and not i32*).
891  std::vector<Value *> dump_params = {_apex_extract_int_arg_loadinst};
892  CallInst *lib_extract_int_fcn_callinst =
893  CallInst::Create(_apex_extract_int_fcn, dump_params, "");
894 
895  logPrintFlat("- " + _apex_extract_int_fcn->getGlobalIdentifier() +
896  " call instruction created: ");
897  lib_extract_int_fcn_callinst->dump();
898  lib_extract_int_fcn_callinst->insertBefore(injection_point);

903 void APEXPass::removeUnneededStuff(Module &M) {
904 
905  // Mapping from function to dependency blocks. Blocks belong to function.
906  // We are going to keep these function:blocks.
907  std::map<const Function *, std::set<DependencyBlock>> function_blocks_to_keep;
908 
909  // @queue and @visited are going to be used for BFS search of dependencies.
910  std::vector<DependencyBlock> queue;
911  std::vector<DependencyBlock> visited;
912 
913  logPrint("Checking if we have any branching dependent on the @path:");
914  {
915  // We will use @path_container as a collecting container for new blocks.
916  // We will append these new blocks after we end the iteration over @path_.
917  std::vector<DependencyBlock> path_container;
918 
919  for (auto &path_block : path_) {
920  printPath();
921  // Set of basic blocks that nodes inside @path_node belong to.
922  std::set<const BasicBlock *> block_bbs;
923 
924  bool block_has_inst_in_if_bb = false;
925 
926  logPrintDbg("- investigating block, collecting basic blocks:");
927  for (auto &node : path_block) {
928  const Instruction *node_inst = cast<Instruction>(node->getValue());
929  std::string bb_name = node_inst->getParent()->getName();
930  block_bbs.insert(node_inst->getParent());
931 
932  if (VERBOSE_DEBUG) {
933  logPrintFlat(" [" + bb_name + "]:");
934  node->getValue()->dump();
935  }
936 
937  // Get first 3 chars of the basic block name.
938  // We are looking for "if.then" basic blocks.
939  if (bb_name.substr(0, 3) == "if.") {
940  block_has_inst_in_if_bb = true;
941  }
942  }
943 
944  logPrintDbg(" - collected basic blocks:");
945  for (const auto &bb : block_bbs) {
946  std::string bb_name = bb->getName();
947  logPrintDbg(" - " + bb_name);
948  }
949 
950  // @path_block is good to go, no branching is dependent on it.
951  if (false == block_has_inst_in_if_bb) {
952  logPrintDbg(" - block has no instruction in if.* basic block");
953  continue;
954  }
955  logPrintDbg(" - block has some instruction in if.* basic block, check "
956  "more:");
957 
958  // We know that some instructions from @block_path are in the BB
959  // that is designated as if.*.
960  // Find branch instruction that services this BB and add block
961  // associated with this branch instruction to the @path.
962 
963  // Function that @path_block belongs to.
964  const Function *block_function =
965  cast<Instruction>(path_block.front()->getValue())->getFunction();
966 
967  logPrintDbg(" - block in fcn: " +
968  block_function->getGlobalIdentifier());
969  logPrintDbg(" - checking fcn for br instructions:");
970  for (const auto &BB : *block_function) {
971  for (const auto &I : BB) {
972 
973  if (isa<BranchInst>(I)) {
974  if (VERBOSE_DEBUG) {
975  logPrintFlat(" - ok, we have br inst: ");
976  I.dump();
977  }
978 
979  for (unsigned i = 0; i < I.getNumOperands(); ++i) {
980  Value *op = I.getOperand(i);
981  std::string op_name = op->getName();
982  logPrintDbg(" - op: " + op_name);
983 
984  for (const auto &bb : block_bbs) {
985  // We have branch instruction which can send execution into
986  // block that is currently in @path.
987  // Get block associated with this branch instruction and put
988  // this block into the @path.
989  if (bb == op) {
990  logPrintDbg(" - bb == op");
991 
992  // Find wich block contains our branch instruction.
993  for (const auto &fcn_block :
994  function_dependency_blocks_[block_function]) {
995  for (const auto &node : fcn_block) {
996  Instruction *node_inst =
997  cast<Instruction>(node->getValue());
998  if (node_inst == &I) {
999  logPrintDbg(" - adding block associated with "
1000  "the branch instruction to the @path");
1001  path_container.push_back(fcn_block);
1002  }
1003  }
1004  }
1005  }
1006  }
1007  }
1008  }
1009  }
1010  }
1011  }
1012  path_.insert(path_.end(), path_container.begin(), path_container.end());
1013  printPath();
1014  logPrint("- done");
1015  }
1016 
1017  if (VERBOSE_DEBUG) {
1018  logPrint("\nPrinting @path:");
1019  printPath();
1020  }
1021 
1022  logPrint("\nComputing what dependency blocks we want to keep:");
1023  {
1024  logPrint("- marking every block from @path as to keep");
1025  for (const auto &block : path_) {
1026  // Mark as visited to make sure we do not process this block in BFS.
1027  visited.push_back(block);
1028 
1029  const Instruction *front_inst =
1030  cast<Instruction>(block.front()->getValue());
1031  function_blocks_to_keep[front_inst->getFunction()].insert(block);
1032  }
1033 
1034  logPrint("- setting up initial queue for BFS search");
1035  // Go over @path and figure out if there are any calls outside the @path.
1036  // If there are, put those called blocks for investigation into the @queue.
1037  for (const auto &path_block : path_) {
1038  for (const auto &called_function :
1039  blocks_functions_callgraph_[path_block]) {
1040 
1041  // Is @called_function part of the path?
1042  bool called_function_part_of_path = false;
1043 
1044  for (const auto &b : path_) {
1045  const Instruction *b_inst = cast<Instruction>(b.front()->getValue());
1046  if (b_inst->getFunction() == called_function) {
1047  called_function_part_of_path = true;
1048  break;
1049  }
1050  }
1051 
1052  if (false == called_function_part_of_path) {
1053  for (auto const &called_function_block :
1054  function_dependency_blocks_[called_function]) {
1055  queue.push_back(called_function_block);
1056  }
1057  }
1058  }
1059  }
1060 
1061  logPrint("- running BFS");
1062  // Run BFS from queue and add everything for keeping that is not visited.
1063  while (false == queue.empty()) {
1064  DependencyBlock current = queue.back();
1065  queue.pop_back();
1066  visited.push_back(current);
1067 
1068  // Figure out what is the parent function of @current.
1069  const Instruction *current_inst =
1070  cast<Instruction>(current.front()->getValue());
1071 
1072  // Store @current block associated with its parent function.
1073  function_blocks_to_keep[current_inst->getFunction()].insert(current);
1074 
1075  // Go over functions that are being called from the @current block.
1076  // Add them to the queue if they were not visited already.
1077  for (const auto &called_function : blocks_functions_callgraph_[current]) {
1078  for (const auto &block : function_dependency_blocks_[called_function]) {
1079 
1080  // Check if we already visited @block.
1081  bool block_already_visited = false;
1082  for (const auto &visited_block : visited) {
1083  if (block == visited_block) {
1084  block_already_visited = true;
1085  break;
1086  }
1087  }
1088  if (false == block_already_visited) {
1089  queue.push_back(block);
1090  }
1091  }
1092  }
1093  }
1094  logPrint("- done");
1095  }
1096 
1097  logPrint("\nCollecting everything that we do not want to keep:");
1098  std::set<DependencyBlock> blocks_to_remove;
1099  std::set<const Function *> functions_to_remove;
1100  {
1101  for (const auto &module_function : M.getFunctionList()) {
1102  // We do not investigate protected functions (do not want to remove them).
1103  if (functionIsProtected(&module_function)) {
1104  logPrintDbg("- fnc is protected: " +
1105  module_function.getGlobalIdentifier());
1106  continue;
1107  }
1108 
1109  // Check if @module_function has entry in the @function_blocks_to_keep
1110  // (that is, there is @module_function with some of its blocks in stored
1111  // in the @function_blocks_to_keep).
1112  if (function_blocks_to_keep.count(&module_function) > 0) {
1113  logPrintDbg("- fnc to +++ KEEP +++: " +
1114  module_function.getGlobalIdentifier());
1115 
1116  // Go over all blocks in the @module_function.
1117  for (auto const &block :
1118  function_dependency_blocks_[&module_function]) {
1119  // Check if @block has entry in the
1120  // @function_blocks_to_keep[&module_function]
1121  if (function_blocks_to_keep[&module_function].count(block) > 0) {
1122  // @block is stored and thus we are going to keep it.
1123  logPrintDbg(" - block to +++ KEEP +++");
1124  } else {
1125  // @block is not stored and we should consider to mark it for
1126  // removal.
1127  bool node_has_branch_inst = false;
1128  for (const auto &node : block) {
1129  if (isa<BranchInst>(node->getValue())) {
1130  node_has_branch_inst = true;
1131  }
1132  }
1133  if (node_has_branch_inst) {
1134  // @node has branch instruction, we will not mark this node
1135  // for removal just to be safe.
1136  logPrintDbg(" - block to +++ KEEP +++: has branch inst");
1137  } else {
1138  blocks_to_remove.insert(block);
1139  logPrintDbg(" - block to remove:");
1140  }
1141  }
1142 
1143  if (VERBOSE_DEBUG) {
1144  for (auto const &node_ptr : block) {
1145  logPrintFlat(" ");
1146  node_ptr->getValue()->dump();
1147  }
1148  }
1149  }
1150  } else {
1151  logPrintDbg("- fnc to remove: " +
1152  module_function.getGlobalIdentifier());
1153  functions_to_remove.insert(&module_function);
1154  }
1155  }
1156  logPrint("- done");
1157  }
1158 
1159  logPrint("\nRemoving unwanted blocks:");
1160  {
1161  for (auto const &block : blocks_to_remove) {
1162  // logPrint("Dropping block:"); // dbg
1163 
1164  // Go ahead and remove instructions stored the @block.
1165  for (auto const &node_ptr : block) {
1166  Instruction *inst = cast<Instruction>(node_ptr->getValue());
1167  // inst->dump(); // dbg
1168 
1169  // Watch out for terminators. Do not remove them!
1170  if (inst->isTerminator()) {
1171  // logPrint("IS TERMINATOR!"); // dbg
1172  continue;
1173  }
1174 
1175  if (false == inst->use_empty()) {
1176  // I'm not sure if this is even necessary.
1177  inst->replaceAllUsesWith(UndefValue::get(inst->getType()));
1178  }
1179 
1180  // Do not erase instruction that is inside @target_instructions_.
1181  // We need those instructions intact.
1182  bool inst_is_target = false;
1183  for (const auto &target_inst : target_instructions_) {
1184  if (target_inst == inst) {
1185  inst_is_target = true;
1186  break;
1187  }
1188  }
1189  if (false == inst_is_target) {
1190  // To be sure that we do not erase one of the target instructions.
1191  inst->eraseFromParent();
1192  }
1193  }
1194  // logPrint(""); // dbg
1195  }
1196  logPrint("- done");
1197  }
1198 
1199  logPrint("\nRemoving unwanted functions:");
1200  {
1201  for (auto const &function : functions_to_remove) {
1202  // Need to get non-const Function ptr.
1203  Function *fcn_to_remove = M.getFunction(function->getGlobalIdentifier());
1204 
1205  logPrint("- removing: " + fcn_to_remove->getGlobalIdentifier());
1206 
1207  // Invalidate all function calls to @function. This is to make sure
1208  // that we do not have any dangling function calls to @function
1209  // after we remove @function. This would produce segfault.
1210  fcn_to_remove->replaceAllUsesWith(
1211  UndefValue::get(fcn_to_remove->getType()));
1212 
1213  // Invalidating instructions inside function that is going to be
1214  // removed is probably not needed. Still, better safe than sorry.
1215  for (auto &BB : *fcn_to_remove) {
1216  for (auto &I : BB) {
1217  if (false == I.use_empty()) {
1218  I.replaceAllUsesWith(UndefValue::get(I.getType()));
1219  }
1220  }
1221  }
1222 
1223  // Finally remove @function.
1224  fcn_to_remove->eraseFromParent();
1225  }
1226  }
1227  logPrint("- done");
1228 }
