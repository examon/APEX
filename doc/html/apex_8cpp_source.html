210 int APEXPass::functionRemoveCalls(const Function *F) {
211  int calls_removed = 0;
212 
213  if (nullptr == F) {
214  return -1;
215  }
216 
217  logPrint("- removing function calls to: " + F->getGlobalIdentifier());
218  for (const Use &use : F->uses()) {
219  if (Instruction *UserInst = dyn_cast<Instruction>(use.getUser())) {
220  std::string message = "- in: ";
221  message += UserInst->getFunction()->getName();
222  message += "; removing call instruction to: ";
223  message += F->getName();
224 
225  if (!UserInst->getType()->isVoidTy()) {
226  // NOTE: Currently only removing void returning calls.
227  // TODO: handle non-void returning function calls in the future
228  message += "; removing non-void returning call is not yet supported! ";
229  message += "[ABORT]";
230  logPrint(message);
231  return -1;
232  }
233 
234  logPrint(message);
235 
236  UserInst->eraseFromParent();
237  calls_removed++;
238  } else {
239  return -1;
240  }
241  }
242  return calls_removed;
243 }

328 int APEXPass::createCallGraph(
329  const Module &M, const std::string &root,
330  std::vector<std::pair<Function *, std::vector<Function *>>> &callgraph) {
331  logPrintUnderline("createCallGraph(): Building call graph");
332  Function *root_fcn = M.getFunction(root);
333  if (nullptr == root_fcn) {
334  logPrint("- root function should not be nullptr after collecting [ERROR]");
335  return -1;
336  }
337 
338  { // Construct call graph.
339  std::vector<Function *> queue;
340  queue.push_back(root_fcn);
341 
342  while (false == queue.empty()) {
343  Function *node = queue.back();
344  queue.pop_back();
345 
346  // Find call functions that are called by the node.
347  std::vector<Function *> node_callees;
348  if (functionGetCallees(node, node_callees) < 0) {
349  logPrint("- failed to collect target callees [ERROR]");
350  return -1;
351  }
352 
353  // Make caller->callees pair and save it.
354  std::pair<Function *, std::vector<Function *>> caller_callees;
355  caller_callees.first = node;
356  caller_callees.second = node_callees;
357  callgraph.push_back(caller_callees);
358 
359  // Put callees to queue, they will be examined next.
360  for (Function *callee : node_callees) {
361  queue.push_back(callee);
362  }
363  }
364  }
365  logPrint("- done");
366  return 0;
367 }

394 int APEXPass::findPath(
395  const std::vector<std::pair<Function *, std::vector<Function *>>>
396  &callgraph,
397  const std::string &source, const std::string &target,
398  std::vector<Function *> &final_path) {
399 
400  logPrintUnderline("findPath(): Finding path from @" + source + " to @" +
401  target);
402  std::vector<std::vector<std::string>> queue;
403  std::vector<std::string> v_start;
404  v_start.push_back(source);
405  queue.push_back(v_start);
406 
407  while (false == queue.empty()) {
408  std::vector<std::string> path = queue.back();
409  queue.pop_back();
410 
411  std::string node = path.back();
412 
413  // Found the end. Store function pointers to @final_path.
414  if (node == target) {
415  for (const std::string node_id : path) {
416  for (auto &caller_callees : callgraph) {
417  Function *caller = caller_callees.first;
418  if (caller->getGlobalIdentifier() == node_id) {
419  final_path.push_back(caller);
420  }
421  }
422  }
423  logPrint("- done");
424  return 0;
425  }
426 
427  // Find adjacent/called nodes of @node and save them to @callees.
428  std::vector<Function *> callees;
429  for (auto &caller_callees : callgraph) {
430  if (node == caller_callees.first->getGlobalIdentifier()) {
431  callees = caller_callees.second;
432  }
433  }
434 
435  // Iterate over adjacent/called nodes and add them to the path.
436  for (Function *callee : callees) {
437  std::string callee_id = callee->getGlobalIdentifier();
438  std::vector<std::string> new_path = path;
439  new_path.push_back(callee_id);
440  queue.push_back(new_path);
441  }
442  }
443 
444  logPrint("- Unable to find path from @" + source + " to @" + target +
445  " [ERROR]");
446 
447  return -1;
448 }

462 void APEXPass::dgInit(Module &M, LLVMDependenceGraph &dg) {
463  if (_LOG_VERBOSE) {
464  logPrintUnderline("dgInit(): Building dependence graph");
465  }
466  // The reasoning why everything below needs to be here:
467  //
468  // In order to get data dependencies, I've replicated
469  // what llvm-dg-dump tool is doing, so for details, check:
470  // dg/tools/llvm-dg-dump.cpp
471  LLVMPointerAnalysis *pta = new LLVMPointerAnalysis(&M);
472  pta->run<analysis::pta::PointsToFlowInsensitive>();
473  dg.build(&M, pta);
474  analysis::rd::LLVMReachingDefinitions rda(&M, pta);
475  rda.run<analysis::rd::ReachingDefinitionsAnalysis>();
476  // rda.run<analysis::rd::SemisparseRda>(); // This is alternative to ^
477  LLVMDefUseAnalysis dua(&dg, &rda, pta);
478  dua.run();
479  dg.computeControlDependencies(CD_ALG::CLASSIC);
480  if (_LOG_VERBOSE) {
481  logPrint("- done");
482  }
